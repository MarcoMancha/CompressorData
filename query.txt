SELECT ID_VARIABLE, NAME, VALUE, DATE 
FROM MEASUREMENT
WHERE DATE BETWEEN '2020-09-18 20:11:59.00' AND '2021-09-18 20:11:59.00'
AND ID_MACHINE = 1 

SELECT DISTINCT ID_VARIABLE, NAME
FROM MEASUREMENT
WHERE ID_MACHINE = 1 AND ID_VARIABLE IN (1,4,25,26,28,29,32,33,34,35,36,37,38,39,41,42)

SELECT ID_VARIABLE, NAME, VALUE, DATE, unit
FROM ERM
WHERE DATE BETWEEN '2020-09-01 00:00:00.00' AND '2020-09-15 17:30:00.00'
AND ID_MACHINE = 3 AND ID_VARIABLE IN (113,114,115,116,117,118)
